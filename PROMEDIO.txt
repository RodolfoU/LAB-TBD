CREATE OR REPLACE PROCEDURE promedio(emergencia character varying)
LANGUAGE SQL
AS $$
select v.nombre, avg(puntaje), e.nombre
from ranking r, voluntario v, tarea ta, emergencia e
where r.id_voluntario = v.id
	and ta.id = r.id_tarea
	and ta.id_emergencia = e.id
	and e.nombre = emergencia
group by v.nombre, e.nombre;
$$;



call promedio('emergencia_0');
